# Epic EHR Replica — Approach A (Express + Vanilla JS) Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a fully interactive Epic Hyperdrive EHR replica for NICU workflows using Node.js/Express backend with vanilla HTML/CSS/JS frontend, demonstrating NeoTherm thermal monitoring integration.

**Architecture:** Express serves a REST API backed by SQLite (better-sqlite3). The frontend is a single-page app loaded from `public/index.html` that manages tab/navigation state client-side. All patient data is generated by a seed script on first run.

**Tech Stack:** Node.js, Express, better-sqlite3, vanilla HTML/CSS/JS

**Spec:** `docs/superpowers/specs/2026-03-14-epic-ehr-replica-design.md`

app.get('/api/patients', (req, res) => {
  const { search } = req.query;

  if (search) {

      "SELECT * FROM patients WHERE first_name LIKE ? OR last_name LIKE ? OR mrn LIKE ?"
    ).all(`%${search}%`, `%${search}%`, `%${search}%`);
  } else {
    patients = db.prepare('SELECT * FROM patients').all();

  res.json(patients);

// Full patient record
app.get('/api/patients/:id', (req, res) => {
  const patient = db.prepare('SELECT * FROM patients WHERE id = ?').get(req.params.id);
  if (!patient) return res.status(404).json({ error: 'Patient not found' });
  res.json(patient);
});

// Vitals
app.get('/api/patients/:id/vitals', (req, res) => {
  const hours = parseInt(req.query.hours) || 24;

// MAR entries (join through medications for patient filtering)
app.get('/api/patients/:id/mar', (req, res) => {
  const { date } = req.query;
  let query = `
    SELECT me.*, m.drug_name, m.dose, m.route, m.frequency
    FROM mar_entries me
    JOIN medications m ON me.medication_id = m.id
    WHERE m.patient_id = ? AND m.status = 'Active'
  `;
  const params = [req.params.id];
  if (date) {
    query += ` AND DATE(me.scheduled_time) = ?`;
    params.push(date);
  }
  query += ` ORDER BY me.scheduled_time`;
  const entries = db.prepare(query).all(...params);
  res.json(entries);
});

- [ ] **Step 3: Create app.js with state management and tab/nav logic**

```javascript
const App = {
  state: {
    tabs: [{ id: 'census', type: 'census', label: 'My Patients' }],
    activeTabId: 'census',
    // For patient chart tabs:
    // { id: 'patient-3', type: 'chart', label: 'Williams, B.', patientId: 3, activeNav: 'summary' }
  },

  init() {
    TopBar.render();
    TabBar.render(this.state.tabs, this.state.activeTabId);
    this.renderActiveTab();
  },

  openPatientChart(patient) {
    const tabId = `patient-${patient.id}`;
    const existing = this.state.tabs.find(t => t.id === tabId);
    if (!existing) {
      this.state.tabs.push({
        id: tabId,
        type: 'chart',
        label: `${patient.last_name}, ${patient.first_name[0]}.`,
        patientId: patient.id,
        activeNav: 'summary'
      });
    }
    this.state.activeTabId = tabId;
    TabBar.render(this.state.tabs, this.state.activeTabId);
    this.renderActiveTab();
    // Check for BPA
    this.checkBPA(patient.id);
  },

  switchTab(tabId) {
    this.state.activeTabId = tabId;
    TabBar.render(this.state.tabs, this.state.activeTabId);
    this.renderActiveTab();
  },

  closeTab(tabId) {
    if (tabId === 'census') return; // can't close census
    this.state.tabs = this.state.tabs.filter(t => t.id !== tabId);
    if (this.state.activeTabId === tabId) {
      this.state.activeTabId = this.state.tabs[this.state.tabs.length - 1].id;
    }
    TabBar.render(this.state.tabs, this.state.activeTabId);
    this.renderActiveTab();
  },

  setActiveNav(navItem) {
    const tab = this.state.tabs.find(t => t.id === this.state.activeTabId);
    if (tab && tab.type === 'chart') {
      tab.activeNav = navItem;
      this.renderActiveTab();
    }
  },

  async renderActiveTab() {
    const tab = this.state.tabs.find(t => t.id === this.state.activeTabId);
    const main = document.getElementById('main-content');
    main.innerHTML = '<div class="loading-spinner"></div>';

    if (tab.type === 'census') {
      main.innerHTML = '';
      await PatientListView.render(main);
    } else if (tab.type === 'chart') {
      main.innerHTML = '';
      // Render chart layout: navigator + storyboard + content area
      const chartLayout = document.createElement('div');
      chartLayout.className = 'chart-layout';

      const nav = document.createElement('div');
      nav.className = 'chart-navigator';
      Navigator.render(nav, tab.activeNav);

      const chartMain = document.createElement('div');
      chartMain.className = 'chart-main';

      const storyboardEl = document.createElement('div');
      storyboardEl.className = 'storyboard-container';
      await Storyboard.render(storyboardEl, tab.patientId);

      const contentEl = document.createElement('div');
      contentEl.className = 'chart-content';
      await this.renderChartView(contentEl, tab.patientId, tab.activeNav);

      chartMain.appendChild(storyboardEl);
      chartMain.appendChild(contentEl);
      chartLayout.appendChild(nav);
      chartLayout.appendChild(chartMain);
      main.appendChild(chartLayout);
    }
  },

  async renderChartView(container, patientId, view) {
    const views = {
      summary: SynopsisView,
      notes: NotesView,
      results: ResultsView,
      orders: OrdersView,
      mar: MARView,
      flowsheets: FlowsheetView,
      problems: ProblemListView,
      imaging: ImagingView,
      neotherm: NeoThermView
    };
    const ViewClass = views[view];
    if (ViewClass) {
      await ViewClass.render(container, patientId);
    }
  },

  async checkBPA(patientId) {
    const alerts = await API.patients.bpa(patientId);
    if (alerts.length > 0) {
      BPAModal.show(alerts[0], patientId);
    }
  }
};

document.addEventListener('DOMContentLoaded', () => App.init());
```
